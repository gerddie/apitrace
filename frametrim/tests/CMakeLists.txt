
MACRO(TRACE_TEST name frame orig_img)
  
  ADD_TEST(NAME ${name} COMMAND
    ${CMAKE_CURRENT_SOURCE_DIR}/compare-trace.sh
    ${CMAKE_BINARY_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${name}.test_trace
    ${frame}
    ${orig_img})
ENDMACRO(TRACE_TEST)
